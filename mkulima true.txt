   this.farms = [];

 addFarm(farmId, name, farmer, phone, address) {
   let farm = {
     farmId: farmId,
     name: name,
     farmer: farmer,
     phone: phone,
     address: address
   }
   this.farms.push (farm)
 }
 getFarm(farmId) {
   for(var i=0; i<this.farms.length; i++) {
     let farm = this.farms[i];
     if (farm.farmId == farmId) {
       return farm
     }
   }
 }

   let farm = this.getFarm(farmId)
   let index = this.farms.indexOf(farm);

 }
