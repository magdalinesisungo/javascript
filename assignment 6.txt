class Kiosk {
constructor(fruits){
this.fruits = fruits || [];
}
addFruit(fruit) {
this.fruits.push(fruit);
}

setFruit(fruit) {
    this.addFruit(fruit);
}

getFruits() {
    return this.fruits;
}
}

var kiosk = new Kiosk(['apple','banana','pasion']);
kiosk.setFruit("mango")
console.log(kiosk.getFruits());
